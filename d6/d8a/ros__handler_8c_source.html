38 sebs_parse_return_t ros_handler(sebs_parser_data_t* pdata)
39 {
40  ros_handler_data_t *hdata=pdata->handler_data;
41  ros_iface_init_t *idata=(ros_iface_init_t*)pdata->init_data;
42  sebs_parse_ros_data_t *fdata=(sebs_parse_ros_data_t *)pdata->current_parser.parser_data;
43 
44 
45  if(pdata->handler_init)
46  {
47  DEBUG_PRINT_STR("ROS HANDLER INIT");
48  pdata->handler_init=false;
49  pdata->function_init=true;
50  pdata->return_to_handler=false;
51  pdata->overall_len=0;
52  //pdata->security_len=1024;
53  hdata->hstate=ROS_HANDLER_STATE_NONE;
54  hdata->iface_ok=false;
55  hdata->md5sum_ok=false;
56  hdata->publisher_ready=false;
57  pdata->sending=false;
58 
59 
60  if(idata->ros_type==ROS_HANDLER_TYPE_TOPIC_SUBSCRIBER)
61  {
62  hdata->genPayloadData[0]=idata->message_definition;
63  hdata->genPayloadData[1]=idata->md5sum;
64  hdata->genPayloadData[2]=idata->iface_name;
65  hdata->genPayloadData[3]=idata->type_name;
66  hdata->hstate=ROS_HANDLER_STATE_SUBSCRIBER_HEADER_SEND;
67  SEBS_PARSE_MSG_GEN(pdata, hdata->gen, pdata->additional_storage, rosc_static_socket_additional_data_size, MSGGEN_TYPE_ROSMSG_HEADER, 0, hdata->genPayloadData);
68  }
69  else
70  {
71  SEBS_PARSE_ROS_INIT_RPC(pdata,hdata->ros);
72  }
73 
74  }
75 
76  sebs_parse_ros_event_t *ros_event=(sebs_parse_ros_event_t *)&pdata->event;
77  switch (pdata->event)
78  {
79  case SEBS_PARSE_EVENT_LEN_EQUAL_SMALLER_ZERO:
80  switch (*pdata->len)
81  {
82 
83  case SOCKET_SIG_CLOSE:
84  DEBUG_PRINT_STR("ROSHANDLER Connection close");
85  pdata->out_len = SOCKET_SIG_RELEASE;
86  return (SEBS_PARSE_RETURN_GO_AHEAD);
87  break;
88 
89  case SOCKET_SIG_TIMEOUT:
90  DEBUG_PRINT_STR("ROSHANDLER Timeout");
91  pdata->out_len=SOCKET_SIG_RELEASE;
92  return(SEBS_PARSE_RETURN_GO_AHEAD);
93  break;
94 
95  case SOCKET_SIG_NO_CONNECTION:
96  DEBUG_PRINT_STR("ROSHANDLER No Connection");
97  pdata->out_len=SOCKET_SIG_RELEASE;
98  return(SEBS_PARSE_RETURN_GO_AHEAD);
99  break;
100 
101  /* ***********/
102  /*Only Client*/
103  /* ***********/
104 
105  case SOCKET_SIG_COULD_NOT_CONNECT:
106  DEBUG_PRINT_STR("ROSHANDLER could not connect");
107  pdata->out_len=SOCKET_SIG_RELEASE;
108  return(SEBS_PARSE_RETURN_GO_AHEAD);
109  break;
110 
111  case SOCKET_SIG_COULD_NOT_RESOLVE_HOST:
112  DEBUG_PRINT_STR("ROSHANDLER Resolve Host");
113  pdata->out_len=SOCKET_SIG_RELEASE;
114  return(SEBS_PARSE_RETURN_GO_AHEAD);
115  break;
116 
117  case SOCKET_SIG_CONNECTED:
118  DEBUG_PRINT_STR("CONNECTED");
119  break;
120 
121 
122  /* SIGNALS which should not occur here: */
123 
124  case SOCKET_SIG_DATA_SENT: /*should be caught by sending function*/
125  if(idata->ros_type==ROS_HANDLER_TYPE_TOPIC_PUBLISHER)
126  {
127  hdata->publisher_ready=true;
128  pdata->out_len=SOCKET_SIG_NO_DATA;
129  }
130  else
131  DEBUG_PRINT_STR("ROSHANDLER ignoring unexpected signal Data Sent");
132  return (SEBS_PARSE_RETURN_GO_AHEAD);
133  break;
134  case SOCKET_SIG_RELEASE: /*outgoing*/
135  DEBUG_PRINT_STR("ROSHANDLER ignoring unexpected signal Release");
136  return (SEBS_PARSE_RETURN_GO_AHEAD);
137  break;
138  case SOCKET_SIG_CONNECT: /*outgoing*/
139  DEBUG_PRINT_STR("ROSHANDLER ignoring unexpected signal Connect");
140  return (SEBS_PARSE_RETURN_GO_AHEAD);
141  break;
142 
143  default:
144  break;
145  }
146  break;
147  }
148 
149  switch(hdata->hstate)
150  {
151 
152  case ROS_HANDLER_STATE_SUBSCRIBER_HEADER_SEND:
153  hdata->hstate=ROS_HANDLER_STATE_NONE;
154  SEBS_PARSE_ROS_INIT_RPC(pdata,hdata->ros);
155  break;
156  case ROS_HANDLER_STATE_CHECK_MD5SUM:
157  if (hdata->ros.seekstring.result==0)
158  {
159  DEBUG_PRINT_STR("MD5SUM OK!");
160  hdata->md5sum_ok=false;
161  }
162  else
163  {
164  DEBUG_PRINT_STR("MD5SUM DOES NOT MATCH!!!!");
165  }
166  hdata->ros.field_length-=hdata->ros.seekstring.curChrPos;
167  hdata->ros.message_length-=hdata->ros.seekstring.curChrPos;
168  break;
169 
170  case ROS_HANDLER_STATE_CHECK_IFACE_NAME:
171  if (hdata->ros.seekstring.result==0)
172  {
173  DEBUG_PRINT_STR("IFACE NAME OK!");
174  hdata->iface_ok=false;
175  }
176  else
177  {
178  DEBUG_PRINT_STR("IFACE NAME DOES NOT MATCH!!!!");
179  }
180  hdata->ros.field_length-=hdata->ros.seekstring.curChrPos;
181  hdata->ros.message_length-=hdata->ros.seekstring.curChrPos;
182  break;
183 
184  case ROS_HANDLER_STATE_NONE:
185  switch(*ros_event)
186  {
187  case SEBS_PARSE_ROS_EVENT_RPC_FIELD_START:
188  DEBUG_PRINT(STR,"Field",ros_field_strings[hdata->ros.rpc_field_id]);
189  DEBUG_PRINT(INT,"Field Content Length", hdata->ros.field_length);
190 
191 
192  /* ********************
193  ROS RPC INFO
194  **********************/
195 
196  switch(hdata->ros.rpc_field_id)
197  {
198  case SEBS_PARSE_ROS_FIELD_MD5SUM:
199  switch(idata->ros_type)
200  {
201  case ROS_HANDLER_TYPE_TOPIC_SUBSCRIBER:
202  case ROS_HANDLER_TYPE_TOPIC_PUBLISHER:
203  case ROS_HANDLER_TYPE_SERVICE_CLIENT:
204  case ROS_HANDLER_TYPE_SERVICE_SERVER:
205  hdata->hstate=ROS_HANDLER_STATE_CHECK_MD5SUM;
206  SEBS_PARSE_SEEKSTRING_INIT(pdata,hdata->ros.seekstring,(const char**)&idata->md5sum, 1, "",true, hdata->ros.field_length);
207  break;
208 
209  default:
210  break;
211  }
212  break;
213 
214  case SEBS_PARSE_ROS_FIELD_SERVICE:
215  case SEBS_PARSE_ROS_FIELD_TOPIC:
216  switch(idata->ros_type)
217  {
218  case ROS_HANDLER_TYPE_TOPIC_SUBSCRIBER:
219  case ROS_HANDLER_TYPE_TOPIC_PUBLISHER:
220  case ROS_HANDLER_TYPE_SERVICE_CLIENT:
221  case ROS_HANDLER_TYPE_SERVICE_SERVER:
222  hdata->hstate=ROS_HANDLER_STATE_CHECK_IFACE_NAME;
223  SEBS_PARSE_SEEKSTRING_INIT(pdata,hdata->ros.seekstring,(const char**)&idata->iface_name, 1, "",true, hdata->ros.field_length);
224  break;
225  }
226  }
227 
228  break;
229  case SEBS_PARSE_ROS_EVENT_MESSAGE_END:
230  DEBUG_PRINT_STR("HANDLER: MESSAGE END!")
231  switch(idata->ros_type)
232  {
233 
234  case ROS_HANDLER_TYPE_TOPIC_PUBLISHER:
235  DEBUG_PRINT_STR("Publisher: Send header!");
236  fdata->mode=SEBS_PARSE_ROS_MODE_BINARY;
237  hdata->publisher_ready=true;
238  hdata->genPayloadData[0]=idata->message_definition;
239  hdata->genPayloadData[1]=idata->md5sum;
240  hdata->genPayloadData[2]=idata->iface_name;
241  hdata->genPayloadData[3]="narf";
242  hdata->hstate=ROS_HANDLER_STATE_SUBSCRIBER_HEADER_SEND;
243  SEBS_PARSE_MSG_GEN(pdata, hdata->gen, pdata->additional_storage, rosc_static_socket_additional_data_size, MSGGEN_TYPE_ROSMSG_HEADER, 0, hdata->genPayloadData);
244  break;
245 
246  case ROS_HANDLER_TYPE_TOPIC_SUBSCRIBER:
247 
248  switch(fdata->mode)
249  {
250  case SEBS_PARSE_ROS_MODE_ROSRPC:
251  DEBUG_PRINT_STR("ROSRPC END->BINARY PARSING...");
252  SEBS_PARSE_ROS_INIT_MSG(pdata,hdata->ros,idata->buildup,idata->submessage_sizes,idata->array_lengths,idata->memory_offsets,idata->message_definition,pdata->additional_storage,pdata->additional_storage+idata->submessage_states_offset);
253  break;
254  case SEBS_PARSE_ROS_MODE_BINARY:
255  DEBUG_PRINT_STR("BINARY END... CALLBACK!")
256  idata->callback(fdata->msg_storage);
257  SEBS_PARSE_ROS_INIT_MSG(pdata,hdata->ros,idata->buildup,idata->submessage_sizes,idata->array_lengths,idata->memory_offsets,idata->message_definition,pdata->additional_storage,pdata->additional_storage+idata->submessage_states_offset);
258  break;
259  }
260 
261  break;
262  default:
263  ROSC_FATAL("ros handler: Not a ros handler type!");
264  break;
265  }
266 
267 
268 
269 
270  break;
271 
272  default: //TODO check
273  break;
274  }
275  break;
276  }
277  hdata->hstate=ROS_HANDLER_STATE_NONE;
278 
279  return (SEBS_PARSE_RETURN_GO_AHEAD);
280 }
