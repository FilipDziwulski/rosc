34 sebs_parse_return_t sebs_parse_http(sebs_parser_data_t* pdata)
35 {
36  sebs_parse_http_data_t *fdata=(sebs_parse_http_data_t *)pdata->current_parser.parser_data;
37  if(pdata->function_init)
38  {
39  pdata->function_init=false;
40  fdata->content_length=SEBS_PARSE_UNKNOWN;
41  fdata->descriptor=SEBS_PARSE_UNKNOWN;
42  fdata->substate=SEBS_PARSE_HTTP_SUBSTATE_STATE_NONE;
43 
44  }
45  while (*pdata->len > 0 && pdata->event == SEBS_PARSE_EVENT_NONE)
46  {
47  bool is_field_content = false;
48  /*
49  * Handling substate results
50  */
51  switch (fdata->substate)
52  {
53  case SEBS_PARSE_HTTP_SUBSTATE_CHECK_METHOD:
54  if (fdata->seekstring.result >= 0)
55  {
56  DEBUG_PRINT_STR("->METHOD found");
57  pdata->event = SEBS_PARSE_HTTP_EVENT_METHOD_PARSED;
58  fdata->state = SEBS_PARSE_HTTP_STATE_REQUEST_ACTION;
59  }
60  else
61  {
62  DEBUG_PRINT_STR("ERROR! HTTP Method unknown");
63  pdata->event =
64  SEBS_PARSE_HTTP_EVENT_ERROR_METHOD_NOT_ALLOWED;
65  }
66  break;
67 
68  case SEBS_PARSE_HTTP_SUBSTATE_CHECK_ACTION:
69  if (fdata->seekstring.result >= 0)
70  {
71  DEBUG_PRINT_STR("->ACTION found...");
72  pdata->event = SEBS_PARSE_HTTP_EVENT_ACTION_PARSED;
73  fdata->state = SEBS_PARSE_HTTP_STATE_REQUEST_HTTP_VER;
74  }
75  else
76  {
77  DEBUG_PRINT_STR("ERROR! HTTP Action unknown");
78  pdata->event =
79  SEBS_PARSE_HTTP_EVENT_ERROR_ACTION_NOT_FOUND;
80  }
81  break;
82 
83  case SEBS_PARSE_HTTP_SUBSTATE_CHECK_REQUEST_HTTP_VER:
84  if (fdata->seekstring.result == HTTP_VAL_HTTP1_0
85  || fdata->seekstring.result == HTTP_VAL_HTTP1_1)
86  {
87  DEBUG_PRINT_STR("Version found...");
88  fdata->state = SEBS_PARSE_HTTP_STATE_HEADLINE_WAIT_END;
89  }
90  else
91  {
92  DEBUG_PRINT_STR("ERROR! HTTP Version unknown");
93  pdata->event =
94  SEBS_PARSE_HTTP_EVENT_ERROR_VERSION_NOT_SUPPORTED;
95  }
96  break;
97 
98  case SEBS_PARSE_HTTP_SUBSTATE_CHECK_RESPONSE_HTTP_VER:
99  if (fdata->seekstring.result == HTTP_VAL_HTTP1_0
100  || fdata->seekstring.result == HTTP_VAL_HTTP1_1)
101  {
102  DEBUG_PRINT_STR("Version found...");
103  fdata->state = SEBS_PARSE_HTTP_STATE_RESPONSE_CODE;
104  }
105  else
106  {
107  DEBUG_PRINT_STR("ERROR! RESPONSE HTTP Version unknown");
108  pdata->event =
109  SEBS_PARSE_HTTP_EVENT_ERROR_VERSION_NOT_SUPPORTED;
110  }
111  break;
112 
113  case SEBS_PARSE_HTTP_SUBSTATE_CHECK_RESPONSE_CODE:
114  switch (fdata->numberparse.result)
115  {
116  case SEBS_PARSE_NUMBERPARSE_ANOTHER_CHAR:
117  if (**pdata->buf == ' ')
118  {
119  DEBUG_PRINT(INT,"Code of HTTP Response", fdata->numberparse.number);
120  pdata->event =
121  SEBS_PARSE_HTTP_EVENT_RESPONSE_CODE;
122  fdata->state = SEBS_PARSE_HTTP_STATE_RESPONSE_STRING;
123  }
124  else
125  {
126  pdata->event =
127  SEBS_PARSE_HTTP_EVENT_ERROR_BAD_RESPONSE;
128  }
129  break;
130 
131  case SEBS_PARSE_NUMBERPARSE_ERROR_NONUMBER:
132  pdata->event =
133  SEBS_PARSE_HTTP_EVENT_ERROR_BAD_RESPONSE;
134  break;
135 
136  case SEBS_PARSE_NUMBERPARSE_MAX_FIGURES:
137  pdata->event =
138  SEBS_PARSE_HTTP_EVENT_ERROR_BAD_RESPONSE;
139  break;
140  }
141  break;
142 
143  case SEBS_PARSE_HTTP_SUBSTATE_CHECK_DESCRIPTOR_ID:
144  if (fdata->seekstring.result > 0)
145  {
146  DEBUG_PRINT(INT,"RESULT",fdata->seekstring.result);
147  DEBUG_PRINT(STR,"Descriptor", fdata->seekstring.stringlist[0]);
148  fdata->descriptor = fdata->seekstring.result;
149  }
150  else
151  {
152  DEBUG_PRINT_STR("Unknown Descriptor...");
153  }
154  fdata->state = SEBS_PARSE_HTTP_STATE_DESCRIPTOR_FIELD_SEPARATOR;
155  break;
156 
157  default:
158  break;
159  }
160 
161  /*
162  * Check if previous state was a substate
163  */
164  if (fdata->substate != SEBS_PARSE_HTTP_SUBSTATE_STATE_NONE)
165  {
166  fdata->substate = SEBS_PARSE_HTTP_SUBSTATE_STATE_NONE;
167  }
168  else
169  {
170  /*
171  * Parsing
172  */
173  switch (**pdata->buf)
174  {
175  case '/':
176  switch (fdata->state)
177  {
178  case SEBS_PARSE_HTTP_STATE_REQUEST_ACTION:
179  fdata->substate = SEBS_PARSE_HTTP_SUBSTATE_CHECK_ACTION;
180  ++*pdata->buf;
181  --*pdata->len;
182  SEBS_PARSE_SEEKSTRING_INIT(pdata,
183  fdata->seekstring,
184  fdata->actions, fdata->actions_len,
185  " ", true,0)
186  ;
187  break;
188 
189  case SEBS_PARSE_HTTP_STATE_FIELD:
190  is_field_content = true;
191  break;
192 
193  case SEBS_PARSE_HTTP_STATE_FIELD_CONTENT:
194  break;
195 
196  case SEBS_PARSE_HTTP_STATE_RESPONSE_STRING:
197  break;
198 
199  default:
200  pdata->event =
201  SEBS_PARSE_HTTP_EVENT_ERROR_BAD_REQUEST;
202  break;
203  }
204  break;
205 
206  case ':':
207  switch (fdata->state)
208  {
209 
210  case SEBS_PARSE_HTTP_STATE_FIELD_CONTENT:
211  break;
212 
213  case SEBS_PARSE_HTTP_STATE_FIELD:
214  is_field_content = true;
215  break;
216 
217  case SEBS_PARSE_HTTP_STATE_DESCRIPTOR_FIELD_SEPARATOR:
218  fdata->state = SEBS_PARSE_HTTP_STATE_FIELD;
219  break;
220 
221  case SEBS_PARSE_HTTP_STATE_RESPONSE_STRING:
222  break;
223 
224  default:
225  pdata->event =
226  SEBS_PARSE_HTTP_EVENT_ERROR_BAD_REQUEST;
227  break;
228 
229  }
230  break;
231 
232  case ' ':
233  switch (fdata->state)
234  {
235 
236  case SEBS_PARSE_HTTP_STATE_FIELD_CONTENT:
237  fdata->state = SEBS_PARSE_HTTP_STATE_FIELD;
238  break;
239 
240  case SEBS_PARSE_HTTP_STATE_RESPONSE_STRING:
241  break;
242 
243  default:
244  break;
245  }
246  break;
247 
248  case '\n':
249  case '\r':
250  switch (fdata->state)
251  {
252  case SEBS_PARSE_HTTP_STATE_FIELD:
253  case SEBS_PARSE_HTTP_STATE_FIELD_CONTENT:
254  case SEBS_PARSE_HTTP_STATE_HEADLINE_WAIT_END:
255  case SEBS_PARSE_HTTP_STATE_RESPONSE_STRING:
256  fdata->state =
257  SEBS_PARSE_HTTP_STATE_DESCRIPTOR_OR_HEADER_END;
258  break;
259 
260  case SEBS_PARSE_HTTP_STATE_DESCRIPTOR_OR_HEADER_END:
261  pdata->event = SEBS_PARSE_HTTP_EVENT_HEADER_END;
262  return (SEBS_PARSE_RETURN_GO_AHEAD);
263  break;
264 
265  case SEBS_PARSE_HTTP_STATE_RESPONSE_CODE:
266  case SEBS_PARSE_HTTP_STATE_DESCRIPTOR_FIELD_SEPARATOR:
267  default:
268  pdata->event =
269  SEBS_PARSE_HTTP_EVENT_ERROR_BAD_REQUEST;
270  break;
271  }
272  break;
273 
274  //Any other char
275  default:
276  switch (fdata->state)
278  {
279  case SEBS_PARSE_HTTP_STATE_REQUEST_METHOD:
280  fdata->substate = SEBS_PARSE_HTTP_SUBSTATE_CHECK_METHOD;
281  SEBS_PARSE_SEEKSTRING_INIT(pdata,
282  fdata->seekstring, fdata->methods,
283  fdata->methods_len, " /\n.\r", true,0)
284  ;
285  break;
286 
287  case SEBS_PARSE_HTTP_STATE_REQUEST_HTTP_VER:
288  fdata->substate =
289  SEBS_PARSE_HTTP_SUBSTATE_CHECK_REQUEST_HTTP_VER;
290  SEBS_PARSE_SEEKSTRING_INIT(pdata,
291  fdata->seekstring, http_header_stdtext,
292  HTTP_HEADER_STDTEXT_LEN, " \n\r", true,0)
293  ;
294  break;
295 
296  case SEBS_PARSE_HTTP_STATE_RESPONSE_HTTP_VER:
297  fdata->substate =
298  SEBS_PARSE_HTTP_SUBSTATE_CHECK_RESPONSE_HTTP_VER;
299  SEBS_PARSE_SEEKSTRING_INIT(pdata,
300  fdata->seekstring, http_header_stdtext,
301  HTTP_HEADER_STDTEXT_LEN, " \n\r", true,0)
302  ;
303  break;
304 
305  case SEBS_PARSE_HTTP_STATE_RESPONSE_CODE:
306  fdata->substate =
307  SEBS_PARSE_HTTP_SUBSTATE_CHECK_RESPONSE_CODE;
308  SEBS_PARSE_NUMBERPARSE_INIT(
309  pdata,
310  fdata->numberparse, 3, false, 10);
311  break;
312 
313  case SEBS_PARSE_HTTP_STATE_DESCRIPTOR_OR_HEADER_END:
314  fdata->substate =
315  SEBS_PARSE_HTTP_SUBSTATE_CHECK_DESCRIPTOR_ID;
316  SEBS_PARSE_SEEKSTRING_INIT(pdata,
317  fdata->seekstring,
318  fdata->descriptors,
319  fdata->descriptors_len, " :", false,0)
320  ;
321  break;
322 
323  case SEBS_PARSE_HTTP_STATE_FIELD:
324  is_field_content = true;
325  break;
326 
327  default:
328  break;
329  }
330  break;
331  }
332 
333  //special external state for field content
334  //must be external due to it's size and need to be placed in multiple positions
335  if (is_field_content)
336  {
337  DEBUG_PRINT_STR("CONTENT_START");
338  fdata->state = SEBS_PARSE_HTTP_STATE_FIELD_CONTENT;
339  pdata->event =
340  SEBS_PARSE_HTTP_EVENT_HEADER_CONTENT;
341  fdata->state = SEBS_PARSE_HTTP_STATE_FIELD_CONTENT;
342  }
343 
344  //We need to exclude header content event here because the first byte of the content
345  //would be skipped for a subfunction called by the handler!
346  if (pdata->len > 0 && pdata->event != SEBS_PARSE_HTTP_EVENT_HEADER_CONTENT)
347  {
348  ++*pdata->buf;
349  --*pdata->len;
350  }
351  else
352  {
353  break;
354  }
355  }
356  }
357  return (SEBS_PARSE_RETURN_GO_AHEAD);
358 }
