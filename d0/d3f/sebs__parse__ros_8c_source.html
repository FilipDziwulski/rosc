520  case SEBS_PARSE_ROSBINARY_SKIP_BYTES: /* skipping bytes of an oversized dynamic array ... */
521  fdata->state=SEBS_PARSE_ROSBINARY_MESSAGE_FIELD;
522  SEBS_PARSE_SKIP_INIT(pdata,fdata->skip,fdata->builtin_array_elements_to_skip);
523  break;
524 
525  case SEBS_PARSE_ROSBINARY_MESSAGE_BUILTIN_ARRAY:
526  {
527 
528  //Setting the pointers to the correct memory location
529  uint32_t *size=(uint32_t*)(fdata->msg_storage+fdata->memory_offset_array[fdata->memory_offset_current_element+1]/*struct base offset*/
530  +fdata->memory_offset_array[fdata->memory_offset_current_element+2]) /*oversize offset*/;
531  bool *oversize=(bool *)(fdata->msg_storage+fdata->memory_offset_array[fdata->memory_offset_current_element+1] /*struct base offset*/
532  +fdata->memory_offset_array[fdata->memory_offset_current_element+3])/*oversize offset*/;
533 
534 
535  fdata->array_length_current_element++; /* next array length */
536  fdata->builtin_is_array=true;
537  if(fdata->buildup_type_array[fdata->buildup_type_current_field]==ROS_MSG_BUILDUP_TYPE_ARRAY_UL)
538  {
539  fdata->builtin_is_dyn_array=true;
540  fdata->memory_offset_current_element+=3; /* increase memory offset to oversize */
541  if(fdata->builtin_array_size>fdata->array_length_array[fdata->array_length_current_element])
542  {
543  *oversize=true;
544  fdata->builtin_array_elements_to_skip=fdata->builtin_array_size-fdata->array_length_array[fdata->array_length_current_element]; /* here we add only the number of entries to skip bytes!!! */
545  fdata->builtin_array_size=fdata->array_length_array[fdata->array_length_current_element];
546  }
547  else
548  {
549  *oversize=false;
550  fdata->builtin_array_elements_to_skip=0;
551  }
552  }
553  else
554  {
555  fdata->builtin_is_dyn_array=false;
556  fdata->memory_offset_current_element+=2; /* increase memory offset to size */
557  fdata->builtin_array_size=fdata->array_length_array[fdata->array_length_current_element];
558  fdata->builtin_array_elements_to_skip=0;
559  }
560 
561  *size=fdata->builtin_array_size;
562  fdata->state=SEBS_PARSE_ROSBINARY_MESSAGE_FIELD;
563  fdata->buildup_type_current_field++; /* next array length, Next buildup field for type .... */
564  }
565  break;
566 
567  default: /* TODO check */
568  break;
569  }
570 
571  if (skipchar)
572  {
573  --fdata->field_length;
574  --fdata->message_length;
575  ++*pdata->buf;
576  --*pdata->len;
577 
578  if (fdata->message_length == 0
579  && fdata->state != SEBS_PARSE_ROSPRC_MESSAGE_LENGTH)
580  {
581 
582  pdata->event = SEBS_PARSE_ROS_EVENT_MESSAGE_END;
583  DEBUG_PRINT_STR("ROSRPC MESSAGE END!");
584  }
585  skipchar = false;
586  }
587 
588  if (pdata->event != SEBS_PARSE_EVENT_NONE)
589  break;
590  }
591  return (SEBS_PARSE_RETURN_GO_AHEAD);
592 }
