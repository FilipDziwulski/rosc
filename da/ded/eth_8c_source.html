90 bool iface_listen( iface_t *iface, port_t *port_number)
91 {
92 
93  listen_socket_t *cur=listen_socket_list_start;
94  while(1)
95  {
96  if(cur->next==0)break;
97  if(cur->state!=LISTEN_SOCKET_STATE_ACTIVE) break;
98  cur=cur->next;
99  }
100 
101  if(cur->state!=LISTEN_SOCKET_STATE_ACTIVE)
102  {
103  listen_socket_id_t sock;
104  sock=abstract_start_listening_on_port(port_number);
105 
106  if(sock==-1)
107  {
108  ROSC_ERROR("Could not open socket!");
109  return(false);
110  }
111 
112  cur->id=sock;
113  cur->interface=iface;
114  DEBUG_PRINT(INT,"Port opened",*port_number)
115  cur->port=*port_number;
116  cur->state=LISTEN_SOCKET_STATE_ACTIVE;
117  }
118  else
119  {
120  return (false);
121  }
122  return (true);
123 }

235 bool rosc_hostlist_resolve(const char* hostname, size_t size, ip_address_ptr ip)
236 {
237  bool nfound=true;
238  int e;
239  for (e = 0; e < rosc_static_lookup_table_size; ++e)
240  {
241  int c;

243  {
244  if((rosc_static_lookup_table+e)->hostname[c]!=hostname[c])
245  {
246  break;
247  }
248  }
249 
250  if(c==size

252  {
253  nfound=false;
254  break;
255  }
256  }
257 
258  if(nfound)

261  }
262  else
263  {
264  int p;
265

267  {
268  ip[p]=(rosc_static_lookup_table+e)->ip[p];
269  }
270 
271  }
272  return (nfound);
